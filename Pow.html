# C:\Scripts\AutoMoveToZ.ps1 $SourceFolder = "C:\Users\$env:USERNAME\Downloads" # BC'nin kaydettiği yer $ZLabels = "Z:\Labels" # Hedef Write-Host "BC → Z:\Labels otomatik taşıma başladı..." -ForegroundColor Green while ($true) { # Yeni CSV dosyalarını bul $files = Get-ChildItem -Path $SourceFolder -Filter "*.csv" | Where-Object { $_.LastWriteTime -gt (Get-Date).AddSeconds(-10) } foreach ($file in $files) { $dest = Join-Path $ZLabels $file.Name try { Copy-Item $file.FullName $dest -Force Write-Host "$(Get-Date -Format 'HH:mm:ss') ✅ $($file.Name) → Z:\Labels (Print başladı)" -ForegroundColor Green # İsteğe bağlı: Kaynak dosyayı sil # Remove-Item $file.FullName } catch { Write-Host "❌ Hata: $_" -ForegroundColor Red } } Start-Sleep -Milliseconds 500 }
